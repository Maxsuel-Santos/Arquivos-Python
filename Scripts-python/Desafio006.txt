print('=== Dobro, Triplo e Raíz ===')
valor = int(input('Digite um valor inteiro: '))
dobro = valor * 2
triplo = valor * 3
raiz = valor ** (1/2)
print('---------------------------------------')
print('O dobro de {} é: {}.\n O triplo de {} é: {}.\n A raíz quadrada de {} é: {:.3f}'.format(valor, dobro, valor, triplo, valor, raiz))
