print('=== PAR ou ÍMPAR? ===')
print('--------------------------')
num = int(input('Informe um número inteiro: '))
if num % 2 == 0:
    print(f'O número {num} é PAR.')
    print('-=-' * 10)
else:
    print(f'O número {num} é ÍMPAR.')
    print('-=-' * 10)
print('--------------------------------')
print('-_-_-_-_-_FIM-_-_-_-_-_-')
